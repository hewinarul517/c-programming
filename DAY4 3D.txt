#include <stdio.h>
int main() {
    int arr[2][3][2] = {
        { {1, 2}, {3, 4}, {5, 6} }, // First block
        { {7, 8}, {9, 10}, {11, 12} } // Second block
    };
    printf("Elements of the 3D Array:\n");
    for (int i = 0; i < 2; i++) {
        printf("Block %d:\n", i);
        for (int j = 0; j < 3; j++) {
            for (int k = 0; k < 2; k++) {
                printf("%d ", arr[i][j][k]);
            }
            printf("\n");
        }
        printf("\n");
    }
    return 0;
}
