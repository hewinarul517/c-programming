#include <stdio.h>
int findMin(int arr[], int n) {
    int min = arr[0];
    for (int i = 1; i < n; i++) {
        if (arr[i] < min) {
            min = arr[i];
        }
    }
    return min;
}

int main() {
    int arr[] = {5, 2, 7, 6, 1, 9, 3}; 
    int n = sizeof(arr) / sizeof(arr[0]);
    int minimum_element = findMin(arr, n);
    printf("The minimum element in the array is: %d\n", minimum_element);

    return 0;
}
