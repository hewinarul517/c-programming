#include <stdio.h>
int main() {
    int numbers[5]; 
    printf("Enter 5 integer values for the array:\n");
    for (int i = 0; i < 5; i++) {
        printf("Enter element at index %d: ", i);
        scanf("%d", &numbers[i]);
    }
    printf("\nThe elements of the array are: ");
    for (int i = 0; i < 5; i++) {
        printf("%d ", numbers[i]);
    }
    printf("\n");
    printf("The element at index 2 is: %d\n", numbers[2]);
    return 0;
}
