#include <stdio.h>

int main() {
    int num1, num2, num3, min_num;

    printf("Enter three integers: ");
    scanf("%d %d %d", &num1, &num2, &num3);

    min_num = num1;

    if (num2 < min_num) {
        min_num = num2;
    }
    if (num3 < min_num) {
        min_num = num3;
    }

    printf("The minimum number is: %d\n", min_num);

    return 0;
}
