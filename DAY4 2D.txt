#include <stdio.h>

int main() {
         int matrix[3][3] = {
        {1, 2, 3},
        {4, 5, 6},
        {7, 8, 9}
    };

        printf("Elements of the 2D array:\n");
    for (int i = 0; i < 3; i++) { // Loop for rows
        for (int j = 0; j < 3; j++) { // Loop for columns
            printf("%d ", matrix[i][j]); // Access element at row 'i' and column 'j'
        }
        printf("\n");    }

        matrix[1][1] = 99; 
    printf("\nMatrix after changing an element:\n");
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) {
            printf("%d ", matrix[i][j]);
        }
        printf("\n");
    }

    return 0;
}
