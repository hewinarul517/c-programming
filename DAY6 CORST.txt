#include <stdio.h>
#include <string.h> 
int main() {
    char str[100]; 
    char searchChar;
    int count = 0; 
    int i;

    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin); 
    str[strcspn(str, "\n")] = 0; 

    printf("Enter the character to count: ");
    scanf(" %c", &searchChar); 
    for (i = 0; str[i] != '\0'; i++) {
        if (str[i] == searchChar) {
            count++;
        }
    }
    printf("The character '%c' appears %d times in the string.\n", searchChar, count);

    return 0;
}
