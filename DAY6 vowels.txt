#include <stdio.h>
#include <string.h>
int isVowel(char c) {
    if (c >= 'A' && c <= 'Z') {
        c = c + 32; 
    }

    if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u') {
        return 1; // It's a vowel
    } else {
        return 0; // It's not a vowel
    }
}

int main() {
    char str[100];    int vowelCount = 0;
    int i;

    printf("Enter a string: ")
    fgets(str, sizeof(str), stdin);    
    str[strcspn(str, "\n")] = 0;


    for (i = 0; str[i] != '\0'; i++) {
        if (isVowel(str[i])) {
            vowelCount++;
        }
    }

    printf("Number of vowels in the string: %d\n", vowelCount);

    return 0;
}
