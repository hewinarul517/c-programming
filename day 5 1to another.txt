#include <stdio.h>
int main() {
    int sourceArray[] = {10, 20, 30, 40, 50};
    int n = sizeof(sourceArray) / sizeof(sourceArray[0]);
    int destinationArray[n];
    for (int i = 0; i < n; i++) {
        destinationArray[i] = sourceArray[i];
    }
    printf("Elements of destinationArray: ");
    for (int i = 0; i < n; i++) {
        printf("%d ", destinationArray[i]);
    }
    printf("\n");
    return 0;
}
