#include <stdio.h>

int main() {
    char name[50];
    int roll_number; 
    int sub1, sub2, sub3, sub4, sub5;
    int total_marks;
    float percentage;
    printf("Enter student name: ");
    scanf("%s", name);
    printf("Enter roll number: ");
    scanf("%d", &roll_number);
    printf("Enter marks for Subject 1 (out of 100): ");
    scanf("%d", &sub1);
    printf("Enter marks for Subject 2 (out of 100): ");
    scanf("%d", &sub2);
    printf("Enter marks for Subject 3 (out of 100): ");
    scanf("%d", &sub3);
    printf("Enter marks for Subject 4 (out of 100): ");
    scanf("%d", &sub4);
    printf("Enter marks for Subject 5 (out of 100): ");
    scanf("%d", &sub5);
    total_marks = sub1 + sub2 + sub3 + sub4 + sub5;
    percentage = (float)total_marks / 5.0;
    printf("\n--- Mark Sheet ---\n");
    printf("Name: %s\n", name);
    printf("Roll Number: %d\n", roll_number);
    printf("------------------\n");
    printf("Subject 1 Marks: %d\n", sub1);
    printf("Subject 2 Marks: %d\n", sub2);
    printf("Subject 3 Marks: %d\n", sub3);
    printf("Subject 4 Marks: %d\n", sub4);
    printf("Subject 5 Marks: %d\n", sub5);
    printf("------------------\n");
    printf("Total Marks: %d\n", total_marks);
    printf("Percentage: %.2f%%\n", percentage);

    if (percentage >= 35) {
        printf("Result: Pass\n");
    } else {
        printf("Result: Fail\n");
    }

    return 0;
}
